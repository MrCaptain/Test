%-分类号:44
%-描述:座骑

%------------------------------------------
%Protocol: 44000 获取座骑信息 
%------------------------------------------
c >> s:
s >> c:
    int:8 StCode    无座骑0(后面字段无效) 1成功(后面字段有效)
    int:8  Level    座骑等级
    int:8  Star     座骑星级
    int:32 Exp      座骑经验值
    int:8  Fashion  当前幻化
    int:8  Riding   状态1:骑 0或其他休息
    array { SkillList   当前学习的技能
        int:8  SkillId  技能ID
        int:8   SkillLv  技能等级
    }
    array { FashionList  可以幻化列表
        int:8  FashId   幻化ID
    }
end

%------------------------------------------
%Protocol: 44001 学习/升级技能 
%------------------------------------------
c >> s:
    int:8 SkillId 技能ID
s >> c:
    int:8 Result 0失败; 1成功; 2铜币不足; 3等级不足; 4历练不足; 5所需技能等级不足;
end

%------------------------------------------
%Protocol: 44003 换装
%------------------------------------------
c >> s:
    int:8  FashId  幻化ID
s >> c:
    int:8 Result 0失败; 1成功; 2铜币不足 3:没有幻化
end

%------------------------------------------
%Protocol: 44004 上坐骑(休息)
%------------------------------------------
c >> s:
s >> c:
    int:8 Result 0失败; 1成功; 
end

%------------------------------------------
%Protocol: 44005 下坐骑(休息)
%------------------------------------------
c >> s:
s >> c:
    int:8 Result 0失败; 1成功; 
end

%------------------------------------------
%Protocol: 44006 升星
%------------------------------------------
c >> s:
s >> c:
    int:8 Result   0失败; 1成功; 
    int:8 NewStar  新星级
end
