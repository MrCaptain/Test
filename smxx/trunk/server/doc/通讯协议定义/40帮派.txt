%- 帮派协议定义：
%-     分类号:40
%- 返回码定义：
%-     0: 未定义错误  1：成功
%- 
%------------------------------------------
%- 所有玩家都可以的操作
%------------------------------------------
%Protocol: 40001 查询帮派(分页待定)
%------------------------------------------
c >> s:
    int:8  PageNo      页号
s >> c:
    int:8  CurPageNo    当前页号
    int:8  TotalPage    总页号
    array { GuildList  　 帮派列表
        int:32  GuildId   帮派ID
        string  GuildName 帮派名称
        int:8   CurNum    当前人数
        int:8   MaxNum    上限人数
        int:8   Level     帮派等级
        int:64  Uid    　 帮主ID
        string  Name   　 帮主名字
    }
end

%------------------------------------------
%Protocol: 40002 创建帮派
%------------------------------------------
c >> s:
    string  Name  帮名
    string  Announce 帮派公告
s >> c:
    int:8 Result 返回结果(参考标准定义)
end

end

%------------------------------------------
%Protocol: 40004 退出所在帮派

s >> c:
    int:8 Result 返回结果(参考标准定义)
end

%------------------------------------------
%Protocol: 40005 查询帮派成员
%------------------------------------------
c >> s:
    int:32 GuildId  帮派ID
s >> c:
    int:8  StCode  返回码(1后面参数才发送,否则不发送)
    array { MemList  成员记录
            int:64  Uid    玩家ID
            string  Name   名字
            int:8   Level  等级
            int:8   Career 玩家职业
            int:8   Gender 玩家性别
            int:8   Position 职位
            int:32  Contrib  贡献
            int:32  LastLoginTime  最后登录时间
            int:8   Online   是否在线1在线,0不在线
        }
end

%------------------------------------------
%Protocol: 40006 发起弹劾
%------------------------------------------
c >> s:
s >> c:
    int:8  Result  返回结果(参考标准定义)
end

%------------------------------------------
%Protocol: 40007 弹劾操作
%------------------------------------------
c >> s:
    int:8   Ops  弹劾类型(1:赞成弹劾, 2:反对弹劾)
s >> c:
    int:8  Result  返回结果(参考标准定义)
end

%------------------------------------------
%Protocol: 40008 获取弹劾信息
%------------------------------------------
c >> s:
s >> c:
    int:8  StCode  返回结果(参考标准定义)
    array {  RejectList     弹劾列表
        int:64 Uid          要弹劾的玩家ID
        int:8  Pos          职位
        int:8  State        状态 (0:未弹劾 1:已弹劾 ）
        int:8  AgreeNum     赞成票数
        int:8  DisagreeNum  反对票数
        int:32 RemainTime   剩余时间(秒)
    }
end

%------------------------------------------
