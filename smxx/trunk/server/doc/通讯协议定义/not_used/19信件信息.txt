%%---------------------------------------------
%% 玩家反馈到GM
%%---------------------------------------------
%-玩家反馈
协议号：19010
c >> s:
    int:8 类型
        1 => Bug
        2 => 投诉
        3 => 建议
        4 => 其它
    string 内容

s >> c:
    int:8 状态
        0 => 失败
        1 => 成功

%%---------------------------------------------
% 协议号：19011 获取反馈列表
%%---------------------------------------------
c >> s:
s >> c:
    array{
     int:32 ID
     int:8     类型
     int:8 (1-已回复,0-未回复)
	 array{
	       string： (NICK or GM)
	 	   string:  内容/回复内容
	 	   string: 	日期+时间
	     }
    }

%--------------------------------
%协议号：19050  是否有未读邮件
%--------------------------------
C >> S

	Int:8	结果码（0失败 1成功）
	Int:8	状态（0未读 1已读）

%--------------------------------
%协议号：19051  邮件列表数据
%--------------------------------
C >> S
	Int:16	请求的页码
S >> C
	Int:8	结果码（0失败 1成功）
	Int:16	当前的页码（与请求的页码一样）
	Int:16	最大页码
	array  邮件列表数据
	{
		Int32:	邮件ID
		Int8:	邮件状态（0未读 1已读）
		String：发件人
		String:	主题
		int32:	邮件发送时间戳
	}

%--------------------------------
%协议号：19052  邮件具体内容
%--------------------------------
C >> S
	Int:32	邮件ID
S >> C
	Int:8	结果码（0失败 1成功）
	Int:32	邮件ID
	String:	发件人
	String:	主题
	String:	正文
	array	邮件附件列表物品
	{
		int32:	物品类型ID
		int32:	物品数量
	}
